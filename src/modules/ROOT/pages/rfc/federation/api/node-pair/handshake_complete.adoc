.Used variables
[source,bash]
----
# Base URL of node B api
$API_B_BASE

# Node B nodeID
$NODE_B_ID

# Node B auth token
$TOKEN_B

# Node A auth token
$TOKEN_A
----

.Example request
[source,bash]
----
curl -X POST "$API_B_BASE/federation/nodes/$NODE_B_ID/handshake-complete" \
  -H "authorization: Bearer $TOKEN_B" \
  --header "Content-Type: application/json" \
  --data "{
    \"token\": \"$TOKEN_A\"
  }";
----
