.Used variables
[source,bash]
----
# Base URL of node A api
$API_A_BASE

# Main administrator JWT for node A
$MAIN_JWT_A

# Node A node ID
$NODE_A_ID
----

.Example request
[source,bash]
----
curl -X POST "$API_A_BASE/federation/nodes/$NODE_A_ID/uri" \
  -H "authorization: Bearer $MAIN_JWT_A" \
  --header "Content-Type: application/json";
----
