.Used variables
[source,bash]
----
# Base URL of node B api
$API_B_BASE

# Main administrator JWT for node B
$MAIN_JWT_B

# Node B nodeID
$NODE_B_ID
----

.Example request
[source,bash]
----
curl -X POST "$API_B_BASE/federation/nodes/$NODE_B_ID/pair" \
  -H "authorization: Bearer $MAIN_JWT_B" \
  --header "Content-Type: application/json";
----
