.Used variables
[source,bash]
----
# Base URL of node B api
$API_B_BASE

# Main administrator JWT for node B
$MAIN_JWT_B

# Node pairing URI
$PAIRING_URI
----

.Example request
[source,bash]
----
curl -X POST "$API_B_BASE/federation/nodes" \
  -H "authorization: Bearer $MAIN_JWT_B" \
  --header "Content-Type: application/json" \
  --data "{
    \"pairingURI\": \"$PAIRING_URI\",
  }";
----
