.Used variables
[source,bash]
----
# Base URL of node A api
$API_A_BASE

# Main administrator JWT for node A
$MAIN_JWT_A

# Node B federation API base URL
$API_B_BASE

# Node name
$NODE_NAME
----

.Example request
[source,bash]
----
curl -X POST "$API_A_BASE/federation/nodes" \
  -H "authorization: Bearer $MAIN_JWT_A" \
  --header "Content-Type: application/json" \
  --data "{
    \"baseURL\": \"$API_B_BASE\",
    \"name\": \"$NODE_NAME\"
  }";
----
